d3.chart.csnw.configurable Example

Line-Bar Chart with Labels and Separate Y-Axes


var config = {
  type: 'Values',
  charts: [
    {
      type: 'Bars', 
      dataKey: 'results', 
      displayAdjacent: true,
      labels: {
        offset: 10,
        position: 'top|bottom',
        style: {
          'font-size': '14px'
        }
      }
    },
    {
      type: 'Line',
      dataKey: 'input',
      labels: {
        position: 'top'
      }
    }
  ],
  axes: {
    x: {title: 'X Title'},
    y: {
      scale: {domain: [0, 20000]}, 
      title: 'Y1 Title'
    },
    secondaryY: {
      dataKey: 'input', 
      scale: {domain: [0, 70]}, 
      title: 'Y2 Title'
    }
  },
  legend: {
    type: 'Inset',
    position: {x: 10, y: 0}
  },
  title: 'Chart Title'
};
