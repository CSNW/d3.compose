2. Add Chart

Add chart...

{% highlight js %} var data = [{x: 0, y: 0}, {x: 1, y: 10}, {x: 2, y: 40}, {x: 3, y: 90}]; var chart = d3.select('#chart').chart('Compose', function(data) { return { charts: { results: { type: 'Lines', data: data } } }; }); chart.draw(data); {% endhighlight %}
{% include editor-placeholder.html width=600 height=400 hide_controls=true wrap=true %}

3. Add Component

Add component...

{% highlight js %} var chart = d3.select('#chart').chart('Compose', function(data) { return { charts: { results: {type: 'Lines', data: data} }, components: { yAxis: { type: 'Axis', position: 'left', scale: {domain: [0, 90]} } } }; }); {% endhighlight %}
{% include editor-placeholder.html width=600 height=400 hide_controls=true wrap=true %}

4. Add Scales

Add scales...

{% highlight js %} var chart = d3.select('#chart').chart('Compose', function(data) { var scales = { x: {data: data, key: 'x'} y: {domain: [0, 100]} }; return { charts: { results: {type: 'Lines', data: data, xScale: scales.x, yScale: scales.y} }, components: { xAxis: { type: 'Axis', position: 'bottom', scale: scales.x }, yAxis: { type: 'Axis', position: 'left', scale: scales.y } } }; }); {% endhighlight %}
{% include editor-placeholder.html width=600 height=400 hide_controls=true wrap=true %}

6. Add Legend

Add legend...

{% highlight js %} var chart = d3.select('#chart').chart('Compose', function(data) { var scales = { x: {data: data, key: 'x'} y: {domain: [0, 100]} }; return { charts: { results: {type: 'Lines', data: data, xScale: scales.x, yScale: scales.y} }, components: { xAxis: { type: 'Axis', position: 'bottom', scale: scales.x }, yAxis: { type: 'Axis', position: 'left', scale: scales.y }, legend: { type: 'Legend', position: 'right', charts: ['results'] } } }; }); {% endhighlight %}
{% include editor-placeholder.html width=600 height=400 hide_controls=true wrap=true %}

7. Add Title

Add Title

{% highlight js %} var chart = d3.select('#chart').chart('Compose', function(data) { var scales = { x: {data: data, key: 'x'} y: {domain: [0, 100]} }; return { charts: { results: {type: 'Lines', data: data, xScale: scales.x, yScale: scales.y} }, components: { xAxis: { type: 'Axis', position: 'bottom', scale: scales.x }, yAxis: { type: 'Axis', position: 'left', scale: scales.y }, legend: { type: 'Legend', position: 'right', charts: ['results'] }, title: { type: 'Title', text: 'd3.compose', 'class': 'chart-title-main' } } }; }); {% endhighlight %}
{% include editor-placeholder.html width=600 height=400 hide_controls=true wrap=true %}

8. Switch to Bars

Switch to Bars...

{% highlight js %} var chart = d3.select('#chart').chart('Compose', function(data) { var scales = { x: {data: data, key: 'x', type: 'ordinal', adjacent: true} y: {domain: [0, 100]} }; return { charts: { results: {type: 'Bars', data: data, xScale: scales.x, yScale: scales.y} }, // Everything stays the same... }; }); {% endhighlight %}
